1   package model.element;
2   
3   import java.awt.image.BufferedImage;
4   
5   import model.SpriteSheet;
6   
7   /**
8    * <h1>The Sprite Class.</h1>
9    *
10   * @author Charles Agostini
11   * @version 1.0
12   */
13  public class Sprite {
14  	/**
15  	 * The sprite image.
16  	 */
17  	private BufferedImage image;
18  	/**
19  	 * The element symbol.
20  	 */
21  	private char symbol;
22  	
23  	/**
24  	 * Create a sprite with defined symbol, 
25  	 * x position on the map,  and y position on the map.
26  	 * The image will have default width = 32 and height = 32.
27  	 * @param symbol
28  	 * 			the element symbol.
29  	 * @param xPos
30  	 * 			the x position on the map.
31  	 * @param yPos
32  	 * 			the y position on the map.
33  	 */
34  	public Sprite(char symbol, int xPos, int yPos) {
35  		this(symbol, xPos, yPos, 32, 32);
36  	}
37  
38  	/**
39  	 * Create a sprite with an associated symbol, an x position on the SpriteSheet, a y position on the 
40  	 * SpriteSheet, a width, and a height.
41  	 * @see model.SpriteSheet
42  	 * @param symbol
43  	 * 			the symbol.
44  	 * @param xPos
45  	 * 			the x position on the spriteSheet.
46  	 * @param yPos
47  	 * 			the y position on the spriteSheet.
48  	 * @param width
49  	 * 			the image width.
50  	 * @param height
51  	 * 			the image height.
52  	 */
53  	public Sprite(char symbol, int xPos, int yPos, int width, int height) {
54  		this.setSymbol(symbol);	
55  		this.selectImage(SpriteSheet.get(), xPos, yPos, width, height);
56  	}
57  	
58  	/**
59  	 * Select the image to use.
60  	 * @param spriteSheet
61  	 * 			the SpriteSheet.
62  	 * @param xPos
63  	 * 			the x position.
64  	 * @param yPos
65  	 * 			the y position.
66  	 * @param width
67  	 * 			the image width.
68  	 * @param height
69  	 * 			the image height.
70  	 */
71  	private void selectImage(BufferedImage spriteSheet, int xPos, int yPos, int width, int height) {
72  		BufferedImage image = spriteSheet.getSubimage((xPos * 35), (yPos * 35), width, height);
73  		setImage(image);
74  	}
75  	
76  	/**
77  	 * Gets the image.
78  	 * @return the image.
79  	 */
80  	public BufferedImage getImage() {
81  		return this.image;
82  	}
83  	
84  	/**
85  	 * Gets the symbol.
86  	 * @return the symbol.
87  	 */
88  	public char getSymbol() {
89  		return this.symbol;
90  	}
91  	
92  	/**
93  	 * Sets the image.
94  	 * @param image
95  	 * 			the image.
96  	 */
97  	private void setImage(BufferedImage image) {
98  		this.image = image;
99  	}
100 	
101 	/**
102 	 * Sets the symbol.
103 	 * @param symbol
104 	 * 			the symbol.
105 	 */
106 	private void setSymbol(char symbol) {
107 		this.symbol = symbol;
108 	}
109 }
