1   package view;
2   
3   import java.awt.Color;
4   import java.awt.Font;
5   import java.awt.Graphics;
6   
7   import GameFrame.IGraphicsBuilder;
8   import model.ILorannModel;
9   import model.element.IElement;
10  import model.element.Position;
11  import model.element.mobile.IMonster;
12  /**
13   * <h1>The GraphicsBuilder.java Class.</h1>
14   *
15   * @author Charles Agostini / Anatole Couasnon
16   * @version 1.0
17   */
18  class GraphicsBuilder implements IGraphicsBuilder {
19  	private final ILorannModel lorannModel;
20  
21  	public GraphicsBuilder(final ILorannModel lorannModel) {
22  		this.lorannModel = lorannModel;
23  	}
24  
25  	@Override
26  	public void applyModelToGraphic(final Graphics graphics) {
27  		if(lorannModel.getMap() != null && lorannModel.getPlayer() != null && lorannModel.getMonsters() != null) {
28  	        clearScreen(graphics);
29  	        buildMapGraph(graphics);
30  	        drawMonsters(graphics);
31  	        drawPlayer(graphics);
32  	        drawSpell(graphics);
33  	        drawScore(graphics);
34  	        drawLife(graphics);
35  	        }
36  
37  	}
38  	
39  	/**
40  	 * Clear the screen
41  	 * @param g
42  	 * to paint on the gamePanel
43  	 */
44  	private void clearScreen(Graphics g) {
45  		g.setColor(Color.BLACK);
46  		g.fillRect(0, 0, 1000, 800);
47  	}
48  	
49  	/**
50  	 * draw the map
51  	 * @param g
52  	 * to paint on the gamePanel
53  	 */
54  	private void buildMapGraph(Graphics g) {
55  		int x = 0, y = 0;
56  		Position position = new Position(x,y);
57  		IElement element;
58  		
59  		for(y = 0; y < lorannModel.getMap().getHeight(); y++) {
60  			for(x = 0; x < lorannModel.getMap().getWidth(); x++) {
61  				element = lorannModel.getMap().getOnTheMap(new Position(x, y));
62  				if(element.getIsAlive() == true) {
63  				g.drawImage(element.getSelectedImage(), element.getPosition().getX()*50, element.getPosition().getY()*50, 50, 50, null);
64  				}
65  				position.setX(x + 1);
66  			}
67  				position.setY(y + 1);
68  		}
69  	}
70  	
71  	/**
72  	 * draw the player
73  	 * @param g
74  	 * to paint on the gamePanel
75  	 */
76  	private void drawPlayer(Graphics g) {
77  		if (lorannModel.getPlayer().getIsAlive() == true)
78  			g.drawImage(lorannModel.getPlayer().getSelectedImage(), lorannModel.getPlayer().getPosition().getX()*50, lorannModel.getPlayer().getPosition().getY()*50, 50, 50, null);
79  	}
80  	
81  	/**
82  	 * draw the spell
83  	 * @param g
84  	 * to paint on the gamePanel
85  	 */
86  	private void drawSpell(Graphics g) {
87  		if (lorannModel.getSpell().getIsAlive() == true) {
88  			g.drawImage(lorannModel.getSpell().getSelectedImage(), lorannModel.getSpell().getPosition().getX()*50, lorannModel.getSpell().getPosition().getY()*50, 50, 50, null);
89  		}
90  	}
91  	
92  	/**
93  	 * draw all the monsters
94  	 * @param g
95  	 * to paint on the gamePanel
96  	 */
97  	private void drawMonsters(Graphics g) {
98  		if(!lorannModel.getMonsters().isEmpty()) {
99  			for (IMonster monster : lorannModel.getMonsters()) {
100 				if (monster.getIsAlive() == true) {
101 					g.drawImage(monster.getSelectedImage(), monster.getPosition().getX()*50, monster.getPosition().getY()*50, 50, 50, null);
102 				}
103 			}
104 		}
105 	}
106 	
107 	/**
108 	 * draw the score
109 	 * @param g
110 	 * to paint on the gamePanel
111 	 */
112 	private void drawScore(Graphics g) {
113 		Font font = new Font("Courier", Font.BOLD, 30);
114         g.setFont(font);
115         g.setColor(Color.RED);
116         g.drawString("Score : ", 100, 680);
117         font = new Font("Courier", Font.ITALIC, 30);
118         g.setFont(font);
119         g.setColor(Color.RED);
120         g.drawString(String.valueOf(lorannModel.getPlayer().getScore()), 300, 680);
121 	}
122 	
123 	/**
124 	 * draw the life
125 	 * @param g
126 	 * to paint on the gamePanel
127 	 */
128 	private void drawLife(Graphics g) {
129 		Font font = new Font("Courier", Font.BOLD, 30);
130         g.setFont(font);
131         g.setColor(Color.GREEN);
132         g.drawString("Life : ", 650, 680);
133         font = new Font("Courier", Font.ITALIC, 30);
134         g.setFont(font);
135         g.setColor(Color.GREEN);
136         g.drawString(String.valueOf(lorannModel.getPlayer().getLife()), 840, 680);
137 	}
138 }
